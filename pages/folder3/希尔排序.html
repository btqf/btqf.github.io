什么是希尔排序？ 所谓希尔排序就是将较大数据集分割成若干小组（逻辑上分组） ，然后分别对每个小组进行插入排序。在插入排序完成后，减少数组分割的距离，如此循环直至分割距离为1。

# 思路：

  1. 首先设置增量gap为len/2,将每个小组进行插入排序后，再将增量gap/2,如此循环直至增量为1.
  2. 对于插入排序不是很了解的，可以观看本篇文章

# 代码

function shellSort(arr) {
    const len = arr.length
    if (len < 2) return arr
    for (let gap = Math.floor(len / 2); gap >= 1; gap = Math.floor(gap / 2)) {
        for (let i = gap; i < len; i++) {
          // arr[i]为无序集合中的第一个元素，即待插入元素
            insert(arr, i, gap)
        }
    }
    return arr
}

function insert(arr, i, gap) {
    let inserted = arr[i]
    // 此时arr[j]为有序集合的最后一个元素
    let j = i - gap
    for (j; j >= 0 && arr[j] > inserted; j -= gap) {
      // 有序集合中大于inserted的元素，后退gap个位置
        arr[j+gap] = arr[j]
    }
    arr[j+gap] = inserted
}
