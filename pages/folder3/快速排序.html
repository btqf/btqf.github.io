什么是快速排序？ 所谓快速排序就是采用一个基准（即主元），将数组中的元素分割成大于主元的一部分A和小于主元的一部分B。对于主元的子集A和B，分别不断重复递归分割，直到所有子集只剩下一个元素，此时数组已排序完成。

# 思路：

  1. 选择第一个元素为主元privot。
  2. 所有小于主元privot的元素，都移到privot的左边，所有大于privot的元素，都移到privot的右边。
  3. 对privot两边的子集，分别不断重复第一步和第二步，直到所有子集只剩下一个元素为止。

# 代码

function quickSort(arr) {
    let len = arr.length
    if (len <= 1) return arr
    let left = [], right = []
    let privot = arr[0]
    for (let i = 1; i < len; i++) {
        if (arr[i] > privot) {
            right.push(arr[i])
        } else {
            left.push(arr[i])
        }
    }
    return quickSort(left).concat([privot],quickSort(right))
}

优化：利用额外两个数组户会增加空间复杂度，可以通过双指针解决。

function quickSort(arr,begin,end){
    if(begin < end){
        let i = begin;
        let j = end;
        let privot = arr[begin];
        while(i < j){
            while(arr[j] > privot && i < j){
                j --;
            }
            arr[i] = arr[j];
            while(arr[i] < privot && i < j){
                i ++;
            }
            arr[j] = arr[i];
        }
        arr[i] = privot;
        quickSort(arr,begin,i-1);
        quickSort(arr,i+1,end);
    }else{
        return;
    }
}
